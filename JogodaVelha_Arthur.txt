        static string nome1, nome2;

        static char simboloAtual = 'X';

        static void Main(string[] args)
        {
            Iniciar();

        static void Iniciar()
        {
            Console.Write("Insira o nome do Jogador 1: ");
            nome1 = (Console.ReadLine());

        {
            Console.WriteLine("        |        |       ");

            Console.WriteLine($"   {p4}    |    {p5}   |   {p6}   ");

            Console.WriteLine($"   {p7}    |    {p8}   |   {p9}   ");

        static void Jogo()
        {
            bool continua = true;
            do { 
                Console.Clear();
                Tabuleiro();

                Console.WriteLine("");
                Console.Write($"{jogadorAtual}, informe o número da posição que deseja: ");

                    Jogadas++;
                    AtualizaMapa(opcao, simboloAtual);

                    {
                        Console.Clear();
                        Tabuleiro();
                        Console.WriteLine("------------------------------------------");
                        Console.WriteLine("------------------------------------------");
                        Console.ForegroundColor = ConsoleColor.Green;
                        Console.WriteLine(jogadorAtual + " ganhou!!");
                        Console.ResetColor();
                        Console.WriteLine("");
                        continua = false;
                    }
                    else if (Empate())
                    {
                        Console.Clear();
                        Tabuleiro();
                        Console.WriteLine("------------------------------------------");
                        Console.WriteLine("------------------------------------------");
                        Console.ForegroundColor = ConsoleColor.Red;
                        Console.WriteLine("Velha!! Ninguém ganhou.");
                        Console.ResetColor();
                        Console.WriteLine("");
                        continua = false;
                    }
                    else
                    {
                        MudaJogador();
                    }
                }
            } while (continua);
        }

            {
                p6 = op;
            } else if (p == 7)
            {
                p7 = op;

            {

            }else if(v == 2 && p2 == '2')
            {
                return true;
            }else if (v == 3 && p3 == '3')
            {

            }else if (v == 4 && p4 == '4')
            {

        {
            if(Jogadas < 7)
            {

            }

            {

                  (p1 == '1' || p2 == '2' || p3 == '3') && (p1 == p2 || p2 == p3 || p1 == p3) ||
